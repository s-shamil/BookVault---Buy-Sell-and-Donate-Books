create or replace PROCEDURE ADD_USER(
ID IN VARCHAR2,
NAME IN VARCHAR2, 
PHONE IN CHAR, 
EMAIL IN VARCHAR2,
LOCATION IN VARCHAR2,
ZIP IN VARCHAR2,
PASS IN VARCHAR2) IS
    IDX NUMBER;
BEGIN
    IDX:=0;
    FOR R IN (SELECT USER_ID FROM "USER")
    LOOP
        IF(R.USER_ID=ID) THEN
            IDX:=1;
        END IF;
    END LOOP;

    IF (IDX=0) THEN
        
        INSERT INTO "USER"(USER_ID,NAME,PHONE_NUM,EMAIL,CURR_CREDIT,LOCATION,BALANCE,ZIP_CODE,PASSWORD)
        VALUES (ID,NAME,PHONE,EMAIL,500,LOCATION,0,ZIP,PASS);
    END IF;
END;
